56def guess_mimetype(url: str) -> Optional[str]: 

57 """ 

58 Attempt to guess the mimetype for a given url 

59 

60 :param url: 

61 :return: 

62 """ 

63 url_path = Path(url) 

64 mime_type, _ = guess_type(url_path.name, strict=False) 

65 if not mime_type: 

66 mime_type = MIMETYPE_MAP.get(url_path.suffix.lower(), None) 

67 return mime_type

95def sanitize_path(text: str, sub_text: str = "-") -> str: 

96 """ 

97 Strips invalid characters from a local file path component. 

98 

99 :param text: 

100 :param sub_text: 

101 :return: 

102 """ 

103 if os.name == "nt" or platform.system().lower() == "windows": 

104 # just replacing `os.sep` is not enough on Windows 

105 # ref https://github.com/ping/odmpy/issues/30 

106 text = ILLEGAL_WIN_PATH_CHARS_RE.sub(sub_text, text) 

107 

108 text = text.replace(os.sep, sub_text) 

109 # also strip away non-printable chars just to be safe 

110 return "".join(c for c in text if c.isprintable())

125def parse_duration_to_milliseconds(text: str) -> int: 

126 """ 

127 Converts a duration string into milliseconds 

128 

129 :param text: A duration string, e.g. "10:15", "10:15.300", "1:10:15" 

130 :return: 

131 """ 

132 mobj = TIMESTAMP_RE.match(text) 

133 if not mobj: 

134 raise ValueError(f"Invalid timestamp text: {text}") 

135 hours = int(mobj.group("hr") or 0) 

136 minutes = int(mobj.group("min") or 0) 

137 seconds = int(mobj.group("sec") or 0) 

138 milliseconds = int((mobj.group("ms") or "0").ljust(3, "0")) 

139 return hours * 60 * 60 * 1000 + minutes * 60 * 1000 + seconds * 1000 + milliseconds

142def parse_duration_to_seconds(text: str) -> int: 

143 """ 

144 Converts a duration string into seconds 

145 

146 :param text: A duration string, e.g. "10:15", "10:15.300", "1:10:15" 

147 :return: 

148 """ 

149 return round(parse_duration_to_milliseconds(text) / 1000.0)

152def mp3_duration_ms(filename: Path) -> int: 

153 # Ref: https://github.com/ping/odmpy/pull/3 

154 # returns the length of the mp3 in ms 

155 

156 # eyeD3's audio length function: 

157 # audiofile.info.time_secs 

158 # returns incorrect times due to its header computation 

159 # mutagen does not have this issue 

160 audio = MP3(filename) 

161 if not audio.info: 

162 raise ValueError(f"Unable to parse MP3 info from: {filename}") 

163 return int(round(audio.info.length * 1000))

167def slugify(value: str, allow_unicode: bool = False) -> str: 

168 """ 

169 Convert to ASCII if 'allow_unicode' is False. Convert spaces to hyphens. 

170 Remove characters that aren't alphanumerics, underscores, or hyphens. 

171 Convert to lowercase. Also strip leading and trailing whitespace. 

172 """ 

173 if allow_unicode: 

174 value = unicodedata.normalize("NFKC", value) 

175 value = re.sub(r"[^\w\s-]", "", value, flags=re.U).strip().lower() 

176 return re.sub(r"[-\s]+", "-", value, flags=re.U) 

177 value = ( 

178 unicodedata.normalize("NFKD", value).encode("ascii", "ignore").decode("ascii") 

179 ) 

180 value = re.sub(r"[^\w\s-]", "", value).strip().lower() 

181 return re.sub(r"[-\s]+", "-", value)
